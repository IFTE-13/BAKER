- trigger for storing employee dml operation EMPLOYEE_LOG_SEQ

CREATE OR REPLACE TRIGGER EMPLOYEE_OPERATION_LOGS 
AFTER INSERT OR UPDATE OR DELETE ON EMPLOYEE

DECLARE
OPNAME EMP_LOG.OPT_NAME%TYPE;

BEGIN
IF INSERTING THEN
OPNAME := 'INSERT';

ELSIF DELETING THEN
OPNAME := 'DELETE';

ELSIF UPDATING THEN
OPNAME := 'UPDATE';

END IF;

INSERT INTO EMPLOYEE_LOG VALUES(EMPLOYEE_LOG_SEQ.NEXTVAL, USER, OPNAME, SYSDATE);
END;
